Zig exports

var counter: i32 = 0;

pub export fn add(a: i32, b: i32) i32 {
    return a + b;
}

pub export fn getCounter() i32 {
    return counter;
}

pub export fn incrementCounter() i32 {
    counter += 1;
    return counter;
}

pub export fn resetCounter() void {
    counter = 0;
}

export tells Zig to put this function into the wasm module's export table. That is why JavaScript can read it from instance.exports. pub only controls visibility to other Zig code.

Relevant build.zig setup

const target = b.resolveTargetQuery(.{
    .cpu_arch = .wasm32,
    .os_tag = .freestanding,
});

const root_module = b.createModule(.{
    .root_source_file = b.path("src/main.zig"),
    .target = target,
    .optimize = optimize,
});

const wasm = b.addExecutable(.{
    .name = "demo",
    .root_module = root_module,
});

wasm.entry = .disabled;
wasm.rdynamic = true;
  • Targeting wasm32-freestanding gives a small module with no OS assumptions.
  • root_source_file points at the Zig code that becomes the wasm module.
  • entry = .disabled avoids needing a native-style executable entry point.
  • rdynamic = true keeps the exported functions visible for JavaScript.

Build flow

  1. Add exported Zig functions with simple ABI-friendly signatures.
  2. Define a wasm target and module in build.zig.
  3. Install the artifact into a directory with b.addInstallArtifact(...).
  4. Create a named build step so zig build wasm has a stable entrypoint.
  5. Run zig build wasm to generate zig-out/dist/demo.wasm.
  6. Run bun run serve so the browser receives the page, JS loader, and wasm file.
const install_dist = b.addInstallArtifact(wasm, .{
    .dest_dir = .{ .override = .{ .custom = "dist" } },
});

const wasm_step = b.step("wasm", "Build WebAssembly artifact into zig-out/dist");
wasm_step.dependOn(&install_dist.step);

Browser loader

The JS side keeps the integration direct:

const wasmUrl = new URL("../zig-out/dist/demo.wasm", import.meta.url);
const response = await fetch(wasmUrl);
const { instance } = await WebAssembly.instantiateStreaming(response, {});
const { add, getCounter, incrementCounter, resetCounter } = instance.exports;
